    <script>
        // Listado de IDs de los objetos para la alternancia de visibilidad
        const solidosIds = ['cubo', 'esfera', 'cilindro', 'cono'];
        const masterGroup = document.getElementById('master-group');

        /**
         * Maneja la visibilidad de las figuras geométricas y actualiza la UI
         * @param {string} selectedId - Nombre del sólido seleccionado ('cubo', 'esfera', etc.)
         */
        function selectSolido(selectedId) {
            solidosIds.forEach(id => {
                const entity = document.getElementById(`solido-${id}`);
                if (id === selectedId) {
                    entity.setAttribute('visible', 'true');
                } else {
                    entity.setAttribute('visible', 'false');
                }
            });

            // Restablece la escala por defecto de la entidad maestra ('1 1 1' relativa a su escala inicial)
            // para mitigar acumulaciones o deformaciones severas provocadas por el zoom previo
            masterGroup.setAttribute('scale', '0.4 0.4 0.4');

            // Actualización del estado visual dinámico de los botones (.active)
            const botones = document.querySelectorAll('.btn-solido');
            botones.forEach(btn => {
                if(btn.textContent.toLowerCase().includes(selectedId)) {
                    btn.classList.add('active');
                } else {
                    btn.classList.remove('active');
                }
            });
        }

        /**
         * Disparador de Seguridad para Autoplay de Cámara en Navegadores Web Móviles.
         * Mitiga los bloqueos estrictos de políticas de privacidad de Safari (iOS) y Chrome (Android).
         */
        window.addEventListener('click', () => {
            const scene = document.querySelector('a-scene');
            if (scene && scene.hasLoaded) {
                const video = document.querySelector('video');
                if (video && video.paused) {
                    video.play().catch(err => console.log("Webcam autoplay forzada controlada:", err));
                }
            }
        }, { once: true }); // Se ejecuta una única vez en la primera interacción para no interferir con la app
    </script>
